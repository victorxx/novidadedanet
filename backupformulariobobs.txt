  <script>
    // Função para carregar conteúdo de uma página externa via fetch
    function carregarTabela() {
      fetch('bobs.html') // Caminho do arquivo que contém a tabela
        .then(response => {
          if (!response.ok) throw new Error('Erro ao carregar o conteúdo.');
          return response.text();
        })
        .then(html => {
          const parser = new DOMParser();
          const doc = parser.parseFromString(html, 'text/html');

          // Tenta pegar a tabela
          const tabela = doc.querySelector('table');
          
          if (tabela) {
            // Se encontrou a tabela, insere ela na página
            document.getElementById('tabelaLojas').innerHTML = tabela.outerHTML;
          } else {
            document.getElementById('tabelaLojas').innerText = 'Tabela não encontrada na página.';
          }
        })
        .catch(error => {
          console.error('Erro ao carregar conteúdo da página:', error);
          document.getElementById('tabelaLojas').innerText = 'Não foi possível carregar a tabela. Tente novamente mais tarde.';
        });
    }

    // Chama a função ao carregar a página
    window.onload = carregarTabela;
  </script>
</main>

<script>
  // Elementos do DOM
  const btnAbrir = document.getElementById('btnAbrirForm');
  const overlay = document.getElementById('overlayForm');
  const btnFechar = document.getElementById('btnFecharForm');
  const form = document.getElementById('formLoja');

  // Número do WhatsApp para enviar (inclua o código do país +55 e DDD, exemplo São Paulo 11)
  const whatsappNumero = '5511999999999'; // substitua pelo número real

  // Abrir o formulário popup
  btnAbrir.addEventListener('click', () => {
    overlay.style.display = 'flex';
  });

  // Fechar o formulário popup
  btnFechar.addEventListener('click', () => {
    overlay.style.display = 'none';
  });

  // Fechar se clicar fora do formulário
  overlay.addEventListener('click', e => {
    if (e.target === overlay) {
      overlay.style.display = 'none';
    }
  });

  // Envio do formulário
  form.addEventListener('submit', e => {
    e.preventDefault();

    const nome = document.getElementById('nome').value.trim();
    const telefone = document.getElementById('telefone').value.trim();
    const endereco = document.getElementById('endereco').value.trim();
    const valorFixo = 'R$ 100,00';

    if (!nome || !telefone || !endereco) {
      alert('Por favor, preencha todos os campos.');
      return;
    }

    // Mensagem formatada para WhatsApp
    const mensagem = 
      `Olá! Gostaria de anunciar minha loja no Bob's Brasil.\n` +
      `Nome: ${nome}\n` +
      `Telefone: ${telefone}\n` +
      `Endereço: ${endereco}\n` +
      `Valor do anúncio: ${valorFixo}`;

    // Criar URL para WhatsApp
    const url = `https://wa.me/${whatsappNumero}?text=${encodeURIComponent(mensagem)}`;

    // Abrir WhatsApp em nova aba
    window.open(url, '_blank');

    // Fecha o formulário
    overlay.style.display = 'none';

    // Limpa o formulário
    form.reset();
  });
</script>
